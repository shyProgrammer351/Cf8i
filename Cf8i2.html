알고리즘

프로그램시작 
   점수들 입력 
   점수 정렬 
      마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 
   정렬된 점수 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter08/ScoreSort.c 

	#include <stdio.h> 

	 

	// 프로그램시작 

1	int main() { 

		int score[3] = { 0 }; 

		int index = 0; 

		int last = 0; 

2		int temp = 0; 

		 

		// 점수들 입력  

3		for( index = 0; index < 3; index++ )  

		{ 

			printf( "0점 ~ 100점 사이의 점수를 입력하세요: " ); 

4			scanf( "%d", &score[index] ); 

		} 

		 

		// 점수 정렬 

		// 마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 

5		for ( last = 2; last >= 0; last-- )  

		{ 

			// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

6			for ( index = 0; index < last; index++ )  

			{ 

				// 앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 

7				if ( score[index] > score[index+1] )  

				{ 

8					temp = score[index]; 

9					score[index] = score[index+1];						 

10					score[index+1] = temp;							 

				} 

			} 

		} 

	 

		// 정렬된 점수 출력 

11		for ( index = 0; index < 3; index++ )  

		{ 

12			printf( "%d) %d ", index + 1, score[index] ); 

		} 

	 

		// 프로그램 종료 

13		return 0; 

	}
