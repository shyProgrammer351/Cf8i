문제

미로 맵을 출력하는 문제입니다 이를 해결하는 다음 프로그램을 해석하세요 
                  
  옷            문

알고리즘

프로그램 시작 
   미로 맵을 2차원 배열로 초기화 
   미로 맵의 2차원 배열을 한 칸씩 출력 
      첫째 Wall이면 벽 출력 
      둘째 ' '이면 길 출력 
      셋째 Exit이면 문 출력 
      넷째 Actor이면 옷 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter08/MazeMap.c 

	#include <stdio.h> 

	 

	// 프로그램 시작 

1	int main() { 

		// 미로 맵을 2차원 배열로 초기화 

		int row=0, column=0; 

2		char map[7][9] = { 

		                   { 'W','W','W','W','W','W','W','W','W' }, 

		                   { 'W','A',' ',' ','W',' ',' ',' ','E' }, 

		                   { 'W','W','W',' ','W',' ','W','W','W' }, 

		                   { 'W',' ',' ',' ','W',' ',' ',' ','W' }, 

		                   { 'W',' ','W','W','W','W','W',' ','W' }, 

		                   { 'W',' ',' ',' ',' ',' ',' ',' ','W' }, 

		                   { 'W','W','W','W','W','W','W','W','W' }  

		                 } ; 

				     	 

		// 미로 맵의 2차원 배열을 한 칸씩 출력 

3		for( row = 0; row < 7; row++ )  

		{ 

4			for( column = 0; column < 9; column++ )  

			{ 

				// 첫째 Wall이면 벽 출력 

5				if( map[row][column] == 'W' ) 

6					printf(" \033[44m   \033[0m"); 

				// 둘째 ' '이면 길 출력 

7				else if( map[row][column] == ' ' ) 

8					printf("  "); 

				// 셋째 Exit이면 문 출력 

9				else if( map[row][column] == 'E' ) 

10					printf(" \033[34m문 \033[0m"); 

				// 넷째 Actor이면 옷 출력 

11				else if( map[row][column] == 'A' ) 

12					printf(" \033[31m옷 \033[0m"); 

			} 

13			printf("\n"); 

		} 

		 

		// 프로그램 종료 

14		return 0; 

	}
